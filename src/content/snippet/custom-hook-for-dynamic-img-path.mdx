##

```
...
import IconNames from '...' // Think about like there is an array of icon names that will be used.

export const useIcon = ({iconName: string}) => {

    const Icon = IconNames.filter((icon) => icon.name === iconName)

    if (!Icon) return <FallbackIcon> // JIT

    return <Icon />
}
```
