##

```
'use client';
...
interface PdfViewerProps {
    triggeringNumber: number;
    srcPath: string;
}

const EmbeddedPdfViewer: FC<PdfViewerProps> = ({triggeringNumber, srcPath}) => {
    const [revalidateNumber, setRevalidateNumber] = useState<number>(0);

    useEffect(() => {
        setRevalidateNumber(prevVal => prevVal + 1);
    }, [triggeringNumber])

    // 'triggeringNumber' is our rerendering dependency. can be Math.random() or any other useful logic.

    return <embed src={srcPath} key={revalidateNumber} type="application/pdf">
}
```
